Par ou Ímpar

Script que verifica se um número informado pelo usuário é par ou ímpar.

Código

      
        //exercício: faça uma aplicação que identifique se um dado numero é par ou impar
        
        //permitir o input/entrada de dados do usuário
        const prompt = require('prompt-sync')();
        const numero = parseFloat(prompt('Qual número você deseja saber se é Par ou Ímpar? '));
        
        //definir lógica das funções a serem executadas segundo input/entrada do usuário
        function verificarParImpar(numero) {
          if (typeof numero !== 'number') {
            return "Por favor, insira um número válido.";
          }
          
          if (numero % 2 === 0) {
            return `${numero} é um número par.`;
          } else {
            return `${numero} é um número ímpar.`;
          }
        }
        
        console.log(verificarParImpar(numero));
