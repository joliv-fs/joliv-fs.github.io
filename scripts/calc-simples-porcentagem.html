Calculadora simples + porcentagem

Script que executa operações matemáticas básicas e calcula porcentagens a partir de valores informados pelo usuário.

Código

      
        //exercício: faça uma aplicação que execute as quatro operações básicas e o módulo matemático (porcentagem)
        
        const prompt = require('prompt-sync')();
        
        //acrescentar opção para selecionar operação
        let operação = prompt('Qual operação deseja realizar (+) (-) (*) (/) (%)? ');
        const Valor1 = parseFloat(prompt('Qual é o primeiro valor dessa operação? '));
        const Valor2 = parseFloat(prompt('Qual é o segundo valor dessa operação? '));
        let total = 0; 
        
        /*
        definir função a ser executada segundo input/entrada do usuário
        lógica desta função: se 'operação a' ... prosseguir com; se não prosseguir com 'operação b' ... do contrário 'mensagem de erro'
        */
        function execOperação(operação, Valor1, Valor2) {
            if (operação == "+") {
                return Valor1 + Valor2;
            } else if
                (operação == "-") {
                return Valor1 - Valor2;
            } else if
                (operação == "*") {
                return Valor1 * Valor2;
            } else if
                (operação == "/") {
                return Valor1 / Valor2;
            } else if
                (operação == "%") {
                return Valor1 / Valor2;
            }   else {
                throw new Error('Operação inválida');
            }
          }
          
          console.log('O resultado é', execOperação(operação, Valor1, Valor2))
